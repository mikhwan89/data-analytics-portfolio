      name:  "due_in_days"

    param: { name: "customer_id"  value: "{{ churn_risk.customer_id._value }}"           }
    param: { name: "churn_score"   value: "{{ churn_risk.churn_score._value }}"            }

  }

# dbt/analyses/alerts/marketing_thresholds.yml
# Post-run hook: evaluates KPIs after the daily mart build.
# Fires Slack alerts when metrics cross configured thresholds.
# Set SLACK_WEBHOOK_MARKETING_OPS in dbt Cloud env vars.

version: 2

alerts:

  - name: cac_spike
    description: CAC exceeds 2× the 30-day rolling average
    model: mart_campaign_performance
    schedule: "0 9 * * *"   # 09:00 daily, after dbt run
    condition:
      sql: |
        select channel, cac_usd,
               avg(cac_usd) over (
                 order by date_month
                 rows between 29 preceding and current row
               ) as cac_30d_avg
        from mart_campaign_performance
        where date_month = date_trunc(current_date(), month)
          and cac_usd > cac_30d_avg * 2.0
    channels:
      slack:
        webhook_env: SLACK_WEBHOOK_MARKETING_OPS
        message: |
          :warning: *CAC Spike — {{ channel }}*
          CAC today:   *${{ cac_usd | round(0) }}*
          30-day avg:  ${{ cac_30d_avg | round(0) }}
          Ratio:       {{ (cac_usd / cac_30d_avg) | round(1) }}×

  - name: roas_floor_breach
    description: ROAS drops below 1.5× on any active channel
    model: mart_campaign_performance
    condition:
      sql: |
        select channel, roas, spend_usd
        from mart_campaign_performance
        where date_month = date_trunc(current_date(), month)
          and roas < 1.5 and spend_usd > 500
    channels:
      slack:
        webhook_env: SLACK_WEBHOOK_MARKETING_OPS
        message: |
          :red_circle: *Low ROAS — {{ channel }}*
          ROAS: *{{ roas }}×*  (floor: 1.5×)
          Spend at risk: ${{ spend_usd | round(0) }}
          → Auto-pause script will run at next check.
# scripts/auto_pause_campaigns.py
# Triggered by Airflow after the daily dbt run completes.
# Queries mart_campaign_performance, pauses Google Ads campaigns
# with ROAS below the configured floor, posts a Slack summary.

from google.ads.googleads.client import GoogleAdsClient
from google.cloud import bigquery
import requests, os

ROAS_FLOOR = 1.5
BQ_PROJECT = "prod-warehouse"
SLACK_HOOK  = os.environ["SLACK_WEBHOOK_MARKETING_OPS"]
ADS_CID    = os.environ["GOOGLE_ADS_CUSTOMER_ID"]

def fetch_low_roas(bq):
    return bq.query("""
        select campaign_id, channel, roas, spend_usd
        from `prod-warehouse.mart.mart_campaign_performance`
        where date_month = date_trunc(current_date(), month)
          and roas      < @roas_floor
          and spend_usd > 500    -- ignore micro-tests
    """, job_config=bigquery.QueryJobConfig(
        query_parameters=[
            bigquery.ScalarQueryParameter("roas_floor", "FLOAT64", ROAS_FLOOR)
        ]
    )).result()

def pause_campaign(ads, campaign_id):
    svc      = ads.get_service("CampaignService")
    campaign = ads.get_type("Campaign")
    campaign.resource_name = svc.campaign_path(ADS_CID, campaign_id)
    campaign.status = ads.enums.CampaignStatusEnum.PAUSED
    op = ads.get_type("CampaignOperation")
    op.update.CopyFrom(campaign)
    op.update_mask.paths.append("status")
    svc.mutate_campaigns(customer_id=ADS_CID, operations=[op])

def notify_slack(paused):
    lines = [f":pause_button: *Auto-paused {len(paused)} campaign(s)*"]
    for c in paused:
        lines.append(f"  • `{c['id']}` — ROAS {c['roas']:.1f}× "
                     f"(floor: {ROAS_FLOOR}×) | spend: ${c['spend']:,.0f}")
    requests.post(SLACK_HOOK, json={"text": "\n".join(lines)})

if __name__ == "__main__":
    bq     = bigquery.Client(project=BQ_PROJECT)
    ads    = GoogleAdsClient.load_from_storage()
    paused = []

    for row in fetch_low_roas(bq):
        pause_campaign(ads, row.campaign_id)
        paused.append({"id": row.campaign_id,
                       "roas": row.roas, "spend": row.spend_usd})

    if paused:
        notify_slack(paused)
    else:
        print("All campaigns above ROAS floor — no action taken.")
