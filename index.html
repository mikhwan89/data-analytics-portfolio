-- stg_google_ads.sql
-- Source: ga_events (partitioned by event_date, ~500M rows, grows 1.5M/day)
-- Full table scan ≈ $3/query  →  with partition filter ≈ $0.03/query

{{ config(materialized = 'view') }}

with source as (

  select *
  from {{ source('raw', 'ga_events') }}
  where
    -- PARTITION PRUNING
    -- BigQuery only reads the date partitions matching this filter.
    -- Without it, every query referencing this view scans the full
    -- table — including every incremental mart refresh downstream.
    event_date >= date_sub(current_date(), interval 90 day)
    and event_date  <  current_date()

),

renamed as (
  select
    campaign_id,
    campaign_name,
    event_date                                        as date_day,
    date_trunc(event_date, month)                     as date_month,
    impressions,
    clicks,
    cast(cost_usd as numeric)                         as cost_usd,
    -- derived metrics
    round(safe_divide(clicks, impressions) * 100, 2)  as ctr_pct,
    round(safe_divide(cost_usd, conversions), 2)      as cpa,
    'google_ads'                                      as ad_platform
  from source
)

select * from renamed
-- int_channel_attribution.sql
-- Ephemeral: no storage cost, executes inline during mart builds.
-- Staging views already enforce partition filters — this model
-- inherits cost-efficiency without needing additional WHERE clauses.

with google_monthly as (
  select
    date_month,
    'google_ads'          as channel,
    sum(cost_usd)         as spend_usd,
    sum(conversions)      as paid_conversions
  from {{ ref('stg_google_ads') }}
  group by 1, 2
),

crm_monthly as (
  select
    acquisition_month   as date_month,
    acquisition_channel as channel,
    count(customer_id)  as new_customers,
    sum(total_ltv_usd)  as total_ltv_usd
  from {{ ref('stg_crm_customers') }}
  group by 1, 2
),

joined as (
  select
    coalesce(p.date_month, c.date_month) as date_month,
    coalesce(p.channel,    c.channel)    as channel,
    p.spend_usd,
    c.new_customers,
    c.total_ltv_usd                      as new_revenue_usd,
    -- CAC via reusable macro (see Macro tab)
    {{ calculate_cac('p.spend_usd', 'c.new_customers') }} as cac_usd
  from google_monthly p
  full outer join crm_monthly c
    using (date_month, channel)
)

select * from joined
-- mart_campaign_performance.sql
-- Incremental model: on each dbt run, only rewrites the two most
-- recent month-partitions instead of rebuilding the full table.
-- BigQuery INSERT OVERWRITE replaces exactly those partitions.
-- Full history rebuild: dbt run --full-refresh

{{
  config(
    materialized         = 'incremental',
    partition_by         = {
      'field'       : 'date_month',
      'data_type'   : 'date',
      'granularity' : 'month'
    },
    cluster_by           = ['channel', 'campaign_id'],
    incremental_strategy = 'insert_overwrite'
  )
}}

with source as (

  select * from {{ ref('int_channel_attribution') }}

  {% if is_incremental() %}
    -- Incremental run: scope to current + prior month only.
  -- BigQuery overwrites exactly those two date-month partitions.
  -- Staging partition filters already cap the upstream scan window.
  where date_month >= date_trunc(
    date_sub(current_date(), interval 1 month), month
  )
  {% endif %}

),

enriched as (
  select
    date_month,
    channel,
    sum(spend_usd)        as total_spend,
    sum(new_customers)    as total_customers,
    sum(new_revenue_usd)  as total_revenue,
    {{ calculate_roas('sum(new_revenue_usd)', 'sum(spend_usd)') }}
                           as roas,
    {{ calculate_cac('sum(spend_usd)', 'sum(new_customers)') }}
                           as cac_usd,
    -- rank channels by ROAS within each month-partition
    rank() over (
      partition by date_month
      order by
        safe_divide(sum(new_revenue_usd), sum(spend_usd)) desc
    ) as roas_rank
  from source
  group by 1, 2
)

select * from enriched
{# macros/get_partition_window.sql
   Centralizes the lookback window so full-refresh vs incremental
   automatically gets the right partition scope.
   Full refresh  → scans 90 days of partitions
   Incremental   → scans only the last 3 days   #}

{% macro get_partition_window(
    date_col,
    lookback_days    = 90,
    incremental_days = 3
) %}

  {% if is_incremental() %}
    {{ date_col }} >= date_sub(current_date(), interval {{ incremental_days }} day)
  {% else %}
    {{ date_col }} >= date_sub(current_date(), interval {{ lookback_days }} day)
  {% endif %}

{% endmacro %}

{# Usage in a staging model:
   where {{ get_partition_window('event_date') }}  #}


{# macros/calculate_roas.sql + calculate_cac.sql
   Reusable metric macros — one definition used in every model  #}

{% macro calculate_roas(revenue_col,
                        spend_col,
                        decimals=2) %}
  round(
    cast({{ revenue_col }} as numeric)
    / nullif(cast({{ spend_col }} as numeric), 0),
    {{ decimals }}
  )
{% endmacro %}

{% macro calculate_cac(spend_col,
                       customers_col,
                       decimals=2) %}
  round(
    cast({{ spend_col }} as numeric)
    / nullif(cast({{ customers_col }} as numeric), 0),
    {{ decimals }}
  )
{% endmacro %}
-- stg_combined_ads.sql
-- PARTITION DEEP DIVE: BigQuery has two partition types, each
-- requiring a different filter syntax to trigger partition pruning.
--
-- Type 1 — Physical column:   table partitioned on a DATE column
--           Filter:           WHERE event_date >= ...
--
-- Type 2 — Ingestion-time:    table partitioned on load timestamp
--           Filter:           WHERE _PARTITIONDATE >= ...
--
-- Filtering fb_ads_raw on event_ts (a regular row column) will NOT
-- prune ingestion-time partitions — a common and costly mistake.

{{ config(materialized = 'view') }}

with google as (

  select event_date, campaign_id,
         impressions, clicks, cost_usd, conversions
  from {{ source('raw', 'google_ads_events') }}
  where
    -- Type 1: filter on the physical partition column
    event_date >= date_sub(current_date(),
                    interval {{ var('lookback_days', 90) }} day)

),

facebook as (

  select
    date(event_ts)  as event_date,
    ad_id           as campaign_id,
    reach           as impressions,
    link_clicks     as clicks,
    spend           as cost_usd,
    purchases       as conversions
  from {{ source('raw', 'fb_ads_raw') }}
  where
    -- Type 2: must filter on _PARTITIONDATE pseudo-column.
    -- Using event_ts here scans ALL partitions regardless of
    -- the date range — identical values, completely different cost.
    _PARTITIONDATE >= date_sub(current_date(),
                        interval {{ var('lookback_days', 90) }} day)

),

unioned as (
  select *, 'google_ads'   as ad_platform from google
  union all
  select *, 'facebook_ads' as ad_platform from facebook
)

select * from unioned

Opinions on Data & BI

Professional takes shaped by a decade of hands-on work across BI platforms, data teams, and industries. These are my honest views — not vendor endorsements.

Why Looker Stands Out: A Data Analyst's Journey Through BI Platforms

After a decade working with Tableau, Power BI, Holistics, Looker, Domo, Mode, and QlikView in professional settings, one platform consistently rises above the rest — and the reasons come down to three architectural decisions that most tools still haven't figured out.

—A Diverse Toolkit

Each platform has its unique strengths, and claiming one is superior in all aspects would be inaccurate. Different businesses have varying priorities — some tools excel at stunning visualizations, others at ecosystem integrations, cost-effectiveness, or scalability. That said, from my perspective having worn the hats of data analyst, Head of Data, and VP of Data, Looker consistently stands out, with Holistics as the sole considerable alternative. This comes down to three key differentiators:

1Code-Based Data Model Layer

Looker's LookML and Holistics' AMQL are the key to many downstream benefits. The core idea is a single source of truth — one governed data model layer that all dashboards draw from, ensuring consistency and reliability across the organization. I don't believe this goal is achievable without a code-based approach, for two reasons:

Collaboration

LookML's Git integration enables team members to develop the data model simultaneously, merge changes, and maintain a clear history of who changed what and when. Without this, every analyst ends up building their own version of the data model, leading to the classic problem: the same metric shows different numbers on two dashboards created by two different analysts. Git solves that. Code also allows inline comments that improve cross-team understanding — something a drag-and-drop interface simply can't replicate.

Gatekeeping

Power users given contributor roles can inadvertently connect unvetted sources — an Excel file pulled in here, a personal Google Sheet there. That flexibility is a recipe for accumulating unverified data sources that silently erode data accuracy. Looker's code-based model makes it straightforward to gate changes through a proper review process before anything hits production, the same way software engineers manage code deployments.

And a small but real quality-of-life point: being able to use Ctrl+C, Ctrl+V, Ctrl+F to manage your data model — instead of wrestling with a mouse-driven GUI — makes day-to-day work meaningfully faster and more enjoyable.

2Data Actions

I've watched dashboard users download data, open another platform, and manually upload it again just to act on what the dashboard showed them. It's a workflow that breaks the analytical loop at the worst possible moment.

Looker's Data Actions — with native integrations to Segment, HubSpot, SendGrid, Firebase, Facebook Ads, Google Ads, Zapier, and others — eliminate this entirely. Users can trigger an action directly from a dashboard row: send an email to a churning customer, push a segment to an ad platform, fire a notification. The analysis and the operational response happen in one place.

Beyond efficiency, there's a data security benefit that's often overlooked: eliminating the need for sensitive data to land on employees' laptops, in email attachments, or in shared drives.

3Direct Query Only

Most platforms offer two connection modes: direct query or data extract. Looker supports only direct query — and I consider this a strength, not a limitation. Here's why:

  1. Reproducibility. Every dashboard component is an SQL query. You can view it directly in Looker or in your database query log — making it trivial for other data team members to reproduce results in a Jupyter notebook, another tool, or a pipeline. Extract-based platforms can't offer this cleanly.
  2. No extract lag. When your ETL finishes loading new data into the warehouse, a direct-query dashboard reflects it immediately. Extract-based dashboards make you wait for the next scheduled refresh — the bigger the data, the longer the delay.
  3. No extract vs. direct confusion. In Tableau and Power BI, certain features only work in extract mode, and others only in direct query. Teams end up making compromises or building hybrid setups that are hard to maintain. Looker removes this entirely — there is only one mode.
  4. Publish changes instantly. In Tableau or Power BI, if you want to apply a second improvement right after publishing one, you may need to re-extract data and wait again. In Looker, publish and it's live immediately, every time.
  5. No wasted extract cycles. Rarely-used LookML components only consume compute when actually queried. Extract-based platforms pull and refresh all components on every schedule, even the ones nobody opens — wasting time and money at scale.
  6. No extract failures. I've personally dealt with cryptic extract failures, inconsistencies between desktop and web applications, and data gateway services crashing after a Windows server auto-update mid-refresh. None of these exist in Looker or Holistics.

4AI Readiness

This is the point that I believe will matter most over the next few years. As AI becomes increasingly embedded in analytics workflows, the structure of your data model determines how much context AI can actually work with.

LookML and AMQL are code — plain text files that live in a Git repository. That makes them natively digestible by AI. An LLM can read your entire semantic layer, understand how dimensions and measures are defined, trace relationships between explores, and reason about your business logic — just as it would with any other codebase. GUI-based data models from Tableau, Power BI, or Metabase are proprietary, often binary or JSON blobs, and were never designed to be read by anything other than their own rendering engine. Feeding those to an AI assistant is either impossible or produces shallow results at best.

The Git history dimension makes this even more powerful. Because every change to a LookML or AMQL model is committed with a message, an AI working with your data model doesn't just see the current state — it can access the full evolution of your data model: what changed, when, why, and by whom. That context is invaluable for understanding intent behind specific metric definitions, debugging unexpected results, or generating accurate data documentation automatically.

In contrast, if you've built your dashboards on a GUI-driven platform, that history simply doesn't exist in any structured, AI-accessible form. You might have a changelog in a shared doc somewhere, but it's not connected to the model itself.

As AI-assisted analytics, natural language querying, and automated documentation become standard expectations rather than novelties, teams running Looker or Holistics will have a structural head start. The rest will be retrofitting — or rebuilding from scratch.

After a decade across these platforms, Looker's architecture addresses the most persistent pain points in BI at an organizational level — not just for individual analysts, but for teams managing shared data models at scale. The sole strong contender is Holistics with AMQL, and I'm watching Malloy as a promising open-source alternative. Tableau and Power BI, in my view, are structurally left behind on this front — implementing a proper code-based data model layer would require them to significantly rearchitect their core product.

Yes, this approach introduces a learning curve and reduces some flexibility. But it's a trade-off worth making for the consistency, reliability, and governance that serious data organizations need.

The Modern Data Stack: Why BigQuery + DBT + Looker Is the Architecture I Keep Coming Back To

The modern data stack isn't just a buzzword — it's a genuine architectural shift in how data teams build and operate. After working across traditional warehouses, monolithic ETL pipelines, and legacy BI tools, I've settled on a view: BigQuery, DBT, and Looker (or Holistics.io) represent the cleanest, most scalable pattern available today — and the reasons are structural, not cosmetic.

—The Stack in One Sentence

BigQuery stores and scales your data. DBT transforms it into trusted, documented models. Looker or Holistics turns those models into governed, self-service analytics. Each layer does one thing well. Each layer is replaceable independently. That modularity is the whole point.

Most data stacks I've inherited were built differently: a monolithic ETL tool that both moved and transformed data, a BI tool that also stored its own extract of the data, and no clear separation between raw source, business logic, and presentation. Debugging a metric discrepancy meant tracing logic across three tools in three different languages. With the modern stack, every layer has a clear owner, a clear language, and a clear responsibility.

1BigQuery: The Warehouse That Scales With You

BigQuery sits at the foundation. It's a fully managed, serverless data warehouse — meaning no cluster provisioning, no index tuning, no capacity planning guesswork. You query terabytes the same way you query megabytes.

The separation of compute and storage is key. You pay for the storage you use and the queries you run — not for a cluster that idles overnight. For most analytics workloads, this is dramatically more cost-efficient than traditional provisioned warehouses.

BigQuery also integrates natively with the rest of the GCP ecosystem: Pub/Sub for streaming ingestion, Dataflow for complex pipelines, Looker for direct-query BI, and Vertex AI for ML workflows. If your organization is building toward a data-driven future on GCP, BigQuery is the obvious center of gravity. But the arguments for BigQuery as the warehouse layer generalize to Snowflake or Redshift — the modern data stack pattern holds regardless of which managed warehouse you pick.

2DBT: Bringing Software Engineering to SQL

DBT is where the paradigm shift really becomes visible. Before DBT, SQL transformations lived in stored procedures, inside ETL tools, in ad-hoc scripts, or — most commonly — spread across all three with no coordination. DBT introduces a structured, version-controlled, testable approach to SQL that data engineers had been asking for since the beginning.

One Language, One Place

Every transformation is a .sql file. Your raw source lands in the warehouse, and DBT transforms it in layers — staging models clean and standardize, intermediate models apply business logic, mart models produce BI-ready tables. Anyone on the team can open a model file and immediately understand what it does. There's no proprietary GUI to learn, no drag-and-drop visual you have to reverse-engineer.

Tests as First-Class Citizens

DBT treats data quality tests the same way software engineers treat unit tests: they're built into the project, not an afterthought. You define that a customer ID must be unique, that revenue must never be null, that channel must be one of a set of known values — and DBT enforces those assertions on every run. When something breaks upstream, you catch it at the transformation layer, not when a confused stakeholder asks why two dashboards disagree.

Documentation That Doesn't Rot

DBT generates a full data catalog from the same files that define your models. Column descriptions, lineage graphs, test coverage — all maintained alongside the transformation code. Contrast this with every other documentation approach I've seen: a Confluence page nobody updates, a spreadsheet with a "last updated: 2021" header, or simply nothing. DBT documentation is co-located with the logic, so it either gets updated or the model doesn't merge.

3Looker (or Holistics): The Governed Semantic Layer

DBT gives you clean, tested tables in your warehouse. The final layer is the semantic model — a single definition of your business metrics that every dashboard and ad-hoc query draws from. This is where Looker's LookML and Holistics' AMQL come in.

The core problem they solve: even with clean DBT mart tables, if two analysts write their own SQL queries against those tables, they'll eventually define revenue slightly differently. One includes refunds. One doesn't. The metric diverges. A governed semantic layer forces a single, auditable definition that everyone queries through — not around.

This is also why I believe Looker and Holistics.io are genuinely interchangeable in this stack. Both treat the semantic layer as code — LookML for Looker, AMQL for Holistics. Both support Git-based workflows. Both connect via direct query to the warehouse, not via data extract. The philosophy is identical: define your metrics once, in code, in version control, and serve them to every consumer.

The choice between them often comes down to scale and budget. Looker is enterprise-grade, built for large organizations with complex permission hierarchies and a heavy emphasis on data governance. Holistics is leaner, faster to set up, and more accessible for smaller teams — while still offering the same code-based semantic layer that makes it a serious platform. Either works. The architecture is the same.

4Why the Layers Fit Together So Well

The stack isn't just a collection of good tools — it's a coherent design. Each layer's output is the next layer's input, and the handoffs are clean:

  1. Ingestion tools (Fivetran, Supermetrics, Airbyte) load raw data into BigQuery. No transformation yet — just raw source fidelity.
  2. DBT transforms raw data into mart tables: clean, tested, documented, with a full lineage graph from source to mart.
  3. Looker or Holistics connects directly to those mart tables via the semantic layer — adding business metric definitions, row-level security, and self-service exploration on top of already-trusted data.

Critically, every layer is independently replaceable. Switch from Fivetran to Airbyte — your DBT models don't change. Switch from BigQuery to Snowflake — your LookML connection string changes, your business logic doesn't. Switch from Looker to Holistics — your mart tables stay exactly the same. This is what a well-designed stack looks like: changes at one layer don't cascade unpredictably into other layers.

There's also a cultural benefit that's easy to overlook. When SQL transformations live in Git (DBT) and the semantic layer lives in Git (LookML/AMQL), the entire analytics codebase is reviewable, versionable, and deployable the same way application code is. Data engineers and analytics engineers speak the same language. Pull requests for metric changes go through the same review process as production code. That discipline compounds over time — it's the difference between a data team that operates like a software team and one that operates like a spreadsheet team.

The modern data stack isn't the cheapest option, and it's not the simplest to set up. It requires a team willing to treat data work as engineering work — with version control, testing, documentation, and code review as non-negotiables.

But once it's in place, the return is substantial: faster iteration, fewer metric disputes, cleaner handoffs between roles, and a foundation that scales with your data volume and team size without a rewrite. BigQuery handles the scale. DBT handles the logic. Looker or Holistics handles the governance. That's the stack — and I haven't seen a more coherent alternative for organizations serious about data.
