-- stg_google_ads.sql
-- Standardize raw Google Ads data

with source as (
  select * from {{ ref('raw_google_ads') }}
),

renamed as (
  select
    campaign_id,
    campaign_name,
    cast(date as date)          as date_day,
    date_trunc('month', cast(date as date))
                                   as date_month,
    impressions,
    clicks,
    cast(cost_usd as numeric)      as cost_usd,
    -- derived metrics
    round(clicks::numeric
      / nullif(impressions, 0) * 100, 2)
                                   as ctr_pct,
    round(cost_usd / nullif(conversions, 0), 2)
                                   as cpa,
    'google_ads'                   as ad_platform
  from source
)

select * from renamed
-- int_channel_attribution.sql
-- Merge paid ads + CRM for blended view

with google_monthly as (
  select
    date_month,
    'google_ads' as channel,
    sum(cost_usd) as spend_usd,
    sum(conversions) as paid_conversions
  from {{ ref('stg_google_ads') }}
  group by 1, 2
),

crm_monthly as (
  select
    acquisition_month as date_month,
    acquisition_channel as channel,
    count(customer_id) as new_customers,
    sum(total_ltv_usd) as total_ltv_usd
  from {{ ref('stg_crm_customers') }}
  group by 1, 2
),

joined as (
  select
    coalesce(p.date_month, c.date_month) as date_month,
    coalesce(p.channel, c.channel)       as channel,
    p.spend_usd,
    c.new_customers,
    -- CAC calculation
    round(p.spend_usd
      / nullif(c.new_customers, 0), 2)  as cac_usd
  from google_monthly p
  full outer join crm_monthly c
    using (date_month, channel)
)

select * from joined
-- mart_marketing_overview.sql
-- Executive-level monthly summary

with channel_data as (
  select * from {{ ref('int_channel_attribution') }}
),

monthly_totals as (
  select
    date_month,
    sum(spend_usd)      as total_spend_usd,
    sum(new_customers)  as total_new_customers,
    sum(new_revenue_usd) as total_revenue,
    round(sum(new_revenue_usd)
      / nullif(sum(spend_usd), 0), 2)
                        as blended_roas
  from channel_data
  group by 1
),

with_growth as (
  select
    *,
    -- month-over-month growth
    round(
      (total_revenue - lag(total_revenue)
        over (order by date_month))
      / nullif(lag(total_revenue)
        over (order by date_month), 0) * 100,
    2) as revenue_mom_pct
  from monthly_totals
)

select * from with_growth
order by date_month
{# macros/calculate_roas.sql
   Reusable metric macros for consistency
   across all models #}

{% macro calculate_roas(revenue_col,
                       spend_col,
                       decimals=2) %}

  round(
    cast({{ revenue_col }} as numeric)
    / nullif(
        cast({{ spend_col }} as numeric),
        0),
    {{ decimals }}
  )

{% endmacro %}


{% macro calculate_cac(spend_col,
                      customers_col,
                      decimals=2) %}

  round(
    cast({{ spend_col }} as numeric)
    / nullif(
        cast({{ customers_col }} as numeric),
        0),
    {{ decimals }}
  )

{% endmacro %}


{# Usage in a model:
   {{ calculate_roas('sum(revenue)', 'sum(spend)') }}
   {{ calculate_cac('total_spend', 'new_customers') }}
#}
